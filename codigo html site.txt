    function toggleLike(postId) {
      if (!currentUser) return;
      
      const allPosts = [...posts, ...samplePosts];
      const post = allPosts.find(p => p.id === postId);
      
      if (post) {
        if (post.liked) {
          post.likes = (post.likes || 0) - 1;
          post.liked = false;
        } else {
          if (post.disliked) {
            post.dislikes = (post.dislikes || 0) - 1;
            post.disliked = false;
          }
          post.likes = (post.likes || 0) + 1;
          post.liked = true;
        }
        saveDataToServer();
        
        if (searchActive) {
          handleSearch();
        } else {
          generatePosts();
        }
      }
    }

    function viewNotification(notificationId, postId) {
      // Mark as read
      const notification = notifications.find(n => n.id == notificationId);
      if (notification) {
        notification.read = true;
        saveDataToServer();
      }
      
      // Find the post
      const post = posts.find(p => p.id == postId);
      if (post) {
        // Show the post in feed
        clearSearch();
        showPage('feed');
        // Scroll to the post
        setTimeout(() => {

          if (postElement) {
            postElement.scrollIntoView({ behavior: 'smooth', block: 'center' });
            postElement.classList.add('highlight-post');
            setTimeout(() => postElement.classList.remove('highlight-post'), 2000);
          }
        }, 100);
      }
      
      // Close dropdown
      document.getElementById('notification-dropdown').classList.remove('show');
      updateNotificationsBadge();
    }

    function addComment(postId) {
      if (!currentUser) return;
      
      const commentInput = document.getElementById(`comment-input-${postId}`);
      const commentText = commentInput.value.trim();
      
      if (!commentText) return;
      
      const newComment = {
        id: Date.now(),
        postId: postId,
        author: currentUser.username,
        text: commentText,
        time: 'agora'
      };
      
      comments.unshift(newComment);
      saveDataToServer();
      
      // Update comments count in post
      const post = posts.find(p => p.id == postId);
      if (post) {
        post.commentsCount = comments.filter(c => c.postId == postId).length;
        saveDataToServer();
      }
      
      // Update comments display
      document.getElementById(`comments-list-${postId}`).innerHTML = renderComments(postId);
      commentInput.value = '';
      
      // Notify post author
      if (post && post.author !== currentUser.username) {
        const notification = {
          id: Date.now(),
          userId: post.author,
          postId: post.id,
          author: currentUser.username,
          displayName: currentUser.displayName,
          text: `comentou no seu post: ${commentText.substring(0, 50)}${commentText.length > 50 ? '...' : ''}`,
          time: 'agora',
          read: false
        };
        
        notifications.push(notification);
        saveDataToServer();
        
        if (notification.userId === currentUser.username) {
          updateNotificationsBadge();
        }
      }
    }

    function followUser(username) {
      if (!currentUser || currentUser.following.includes(username)) return;
      
      const user = users.find(u => u.username === username);
      if (!user) return;
      
      // Check if user requires approval for followers
      if (user.followRequests) {
        // Add to follow requests
        if (!user.followRequests.includes(currentUser.username)) {
          user.followRequests.push(currentUser.username);
          saveDataToServer();
          
          // Notify user
          const notification = {
            id: Date.now(),
            userId: username,
            author: currentUser.username,
            displayName: currentUser.displayName,
            text: 'quer seguir voc√™',
            time: 'agora',
            read: false,
            type: 'follow-request'
          };
          notifications.push(notification);
          saveDataToServer();
          
          // Update UI
          if (viewingUserProfile && viewingUserProfile.username === username) {
            viewUserProfile(username);
          }
          
          alert('Solicita√ß√£o de seguimento enviada!');
        }
      } else {
        // Directly follow if no approval needed
        currentUser.following.push(username);
        user.followers.push(currentUser.username);
        saveDataToServer();
        
        // Update UI
        if (viewingUserProfile && viewingUserProfile.username === username) {
          viewUserProfile(username);
        }
      }
    }

    function unfollowUser(username) {
      if (!currentUser || !currentUser.following.includes(username)) return;
      
      currentUser.following = currentUser.following.filter(u => u !== username);
      
      const user = users.find(u => u.username === username);
      if (user) {
        user.followers = user.followers.filter(u => u !== currentUser.username);
      }
      
      saveDataToServer();
      
      // Update UI
      if (viewingUserProfile && viewingUserProfile.username === username) {
        viewUserProfile(username);
      }
    }

    function loadFollowRequests() {
      if (!currentUser) return;
      
      const container = document.getElementById('follow-requests-container');
      
      if (currentUser.followRequests && currentUser.followRequests.length > 0) {
        container.innerHTML = currentUser.followRequests.map(username => {
          const user = users.find(u => u.username === username);
          const avatarStyle = user.avatar ? `background-image: url('${user.avatar}')` : '';
          
          return `
            <div class="follow-request-item">
              <div style="display: flex; align-items: center; gap: 10px;">
                <div class="user-search-avatar" style="${avatarStyle}">${user.avatar ? '' : user.displayName.charAt(0).toUpperCase()}</div>
                <div>
                  <div class="user-search-name">${user.displayName}</div>
                  <div class="user-search-username">@${username}</div>
                </div>
              </div>
              <div class="follow-request-buttons">
                <button class="follow-btn" onclick="acceptFollowRequest('${username}')">Aceitar</button>
                <button class="unfollow-btn" onclick="rejectFollowRequest('${username}')">Recusar</button>
              </div>
            </div>
          `;
        }).join('');
      } else {
        container.innerHTML = '<p>Nenhuma solicita√ß√£o pendente.</p>';
      }
    }

    function acceptFollowRequest(username) {
      if (!currentUser || !currentUser.followRequests.includes(username)) return;
      
      // Remove from requests
      currentUser.followRequests = currentUser.followRequests.filter(u => u !== username);
      
      // Add to followers
      currentUser.followers.push(username);
      
      // Add to following for the requester
      const requester = users.find(u => u.username === username);
      if (requester) {
        requester.following.push(currentUser.username);
      }
      
      saveDataToServer();
      loadFollowRequests();
      
      // Notify requester
      const notification = {
        id: Date.now(),
        userId: username,
        author: currentUser.username,
        displayName: currentUser.displayName,
        text: 'aceitou sua solicita√ß√£o de seguimento',
        time: 'agora',
        read: false
      };
      notifications.push(notification);
      saveDataToServer();
    }

    function rejectFollowRequest(username) {
      if (!currentUser || !currentUser.followRequests.includes(username)) return;
      
      // Remove from requests
      currentUser.followRequests = currentUser.followRequests.filter(u => u !== username);
      
      saveDataToServer();
      loadFollowRequests();
    }

    function showChat() {
      showPage('chat');
      loadConversations();
    }

    function loadConversations() {
      if (!currentUser) return;
      
      const convList = document.getElementById('conversations-list');
      convList.innerHTML = '';
      
      // Agrupa mensagens por usu√°rio
      const userMessages = {};
      conversations.forEach(msg => {
        const outroUsuario = msg.sender === currentUser.username ? msg.receiver : msg.sender;
        if (!userMessages[outroUsuario]) userMessages[outroUsuario] = [];
        userMessages[outroUsuario].push(msg);
      });
      
      // Cria a lista de conversas
      Object.keys(userMessages).forEach(username => {
        const user = users.find(u => u.username === username);
        if (!user) return;
        
        const ultimaMsg = userMessages[username].slice(-1)[0];
        const naoLidas = userMessages[username].filter(m => 
          m.receiver === currentUser.username && !m.read
        ).length;
        
        const avatarStyle = user.avatar ? `background-image: url('${user.avatar}')` : '';
        
        const convItem = document.createElement('div');
        convItem.className = 'user-search-item';
        convItem.onclick = () => openChat(username);
        convItem.innerHTML = `
          <div class="user-search-avatar" style="${avatarStyle}">${user.avatar ? '' : user.displayName.charAt(0).toUpperCase()}</div>
          <div class="user-search-info" style="flex:1;">
            <div style="display:flex;justify-content:space-between;">
              <div class="user-search-name">${user.displayName}</div>
              <div class="post-time">${ultimaMsg.time}</div>
            </div>
            <div style="display:flex;justify-content:space-between;">
              <div class="user-search-username" style="font-size:0.8em;color:#888;max-width:70%;white-space:nowrap;overflow:hidden;text-overflow:ellipsis;">
                ${ultimaMsg.sender === currentUser.username ? 'Voc√™: ' : ''}${ultimaMsg.text}
              </div>
              ${naoLidas > 0 ? `<span style="background:#ff1a1a;color:white;border-radius:50%;width:20px;height:20px;display:flex;align-items:center;justify-content:center;font-size:0.7em;">${naoLidas}</span>` : ''}
            </div>
          </div>
        `;
        convList.appendChild(convItem);
      });
    }

    function openChat(username) {
      currentChat = username;
      document.getElementById('conversations-list').style.display = 'none';
      document.getElementById('chat-container').style.display = 'block';
      
      const user = users.find(u => u.username === username);
      document.getElementById('chat-with-user').textContent = `Conversando com ${user.displayName}`;
      
      loadChatMessages();
    }

    function closeChat() {
      currentChat = null;
      document.getElementById('conversations-list').style.display = 'block';
      document.getElementById('chat-container').style.display = 'none';
    }

    function loadChatMessages() {
      const chatMsgs = document.getElementById('chat-messages');
      chatMsgs.innerHTML = '';
      
      const mensagens = conversations.filter(msg => 
        (msg.sender === currentUser.username && msg.receiver === currentChat) ||
        (msg.sender === currentChat && msg.receiver === currentUser.username)
      ).sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));
      
      mensagens.forEach(msg => {
        const souEu = msg.sender === currentUser.username;
        const user = users.find(u => u.username === msg.sender);
        const avatarStyle = user.avatar ? `background-image: url('${user.avatar}')` : '';
        
        const msgDiv = document.createElement('div');
        msgDiv.style.display = 'flex';
        msgDiv.style.flexDirection = souEu ? 'row-reverse' : 'row';
        msgDiv.style.margin = '10px 0';
        msgDiv.style.gap = '10px';
        
        msgDiv.innerHTML = `
          <div class="user-search-avatar" style="${avatarStyle};width:30px;height:30px;">${user.avatar ? '' : user.displayName.charAt(0).toUpperCase()}</div>
          <div style="max-width:70%;background:${souEu ? '#960e0e' : '#36393f'};padding:8px 12px;border-radius:12px;word-break:break-word;">
            <div style="font-size:0.8em;color:#888;margin-bottom:5px;">${msg.time}</div>
            <div>${msg.text}</div>
          </div>
        `;
        
        chatMsgs.appendChild(msgDiv);
        
        // Marca como lida
        if (msg.receiver === currentUser.username && !msg.read) {
          msg.read = true;
          saveDataToServer();
        }
      });
      
      chatMsgs.scrollTop = chatMsgs.scrollHeight;
    }

    function sendMessage() {
      const input = document.getElementById('chat-message');
      const texto = input.value.trim();
      if (!texto || !currentChat) return;
      
      const novaMsg = {
        id: Date.now(),
        sender: currentUser.username,
        receiver: currentChat,
        text: texto,
        time: 'agora',
        timestamp: Date.now(),
        read: false
      };
      
      conversations.push(novaMsg);
      saveDataToServer();
      input.value = '';
      loadChatMessages();
      loadConversations(); // Atualiza a lista de conversas
    }
